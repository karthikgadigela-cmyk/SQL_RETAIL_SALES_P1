				transactions_id	INT PRIMARY KEY,

				category VARCHAR(20),
				quantiy INT,
				price_per_unit	FLOAT,
				cogs FLOAT,

);

--BUSSINESS ANALYSIS

--1.WRITE SQL QUERY TO RETRIVE ALL COLUMNS FOR SALE MADE ON 20203-11-22
SELECT * 
FROM RETAIL_SALES_ANALYSIS
WHERE 
	SALE_DATE = '2023-11-22';


SELECT CATEGORY , SUM(QUANTIY) AS TOTAL, SUM(TOTAL_SALE)AS TOTAL_AMOUNT
FROM 
	RETAIL_SALES_ANALYSIS
WHERE 
	CATEGORY = 'CLOTHING'
	GROUP BY CATEGORY;
	AND SALE_DATE >= '2022-11-01'
	AND SALE_DATE<= '2022-12-01'
	AND QUANTIY>= 4;

--TOTAL_SALES FOR EACH CATEGORY WITH ORDERS

SELECT CATEGORY,
	SUM(TOTAL_SALE)AS TOTAL_AMOUNT,
	COUNT(*) AS TOTAL_ORDERS
	FROM RETAIL_SALES_ANALYSIS
	GROUP BY CATEGORY
	ORDER BY TOTAL_AMOUNT;


--CUSTOMERS AVG AGE WHO BOUGHT BEAUTY PRODUCTS
SELECT CATEGORY,
	AVG(AGE) AS AVG_AGE
	FROM RETAIL_SALES_ANALYSIS
	WHERE CATEGORY = 'BEAUTY'
	GROUP BY CATEGORY;

--TRANSACTIONS WHICH TOTAL SALES ABOVE 1000

SELECT *
	FROM 
RETAIL_SALES_ANALYSIS
	WHERE TOTAL_SALE>1000;

--NUMBER OF TRANSACTIONS PER CATEGORY BY GENDER

SELECT CATEGORY,
	GENDER,
	COUNT(*) AS TOTAL_TRANSACTIONS
FROM
RETAIL_SALES_ANALYSIS
GROUP BY 
	CATEGORY, GENDER
	ORDER BY TOTAL_TRANSACTIONS;

--AVG SALES OF THE MONTH AND THE BEST MONTH

SELECT * FROM (

SELECT 
		YEAR (SALE_DATE) AS YEAR,
		MONTH(SALE_DATE)AS MONTH,
		AVG(TOTAL_SALE) AS AVG_SALES,
		RANK() OVER(PARTITION BY YEAR (SALE_DATE) ORDER BY AVG(TOTAL_SALE) DESC) AS RNK
		FROM RETAIL_SALES_ANALYSIS
		GROUP BY 
		YEAR (SALE_DATE),
		MONTH(SALE_DATE) 
) AS T1
WHERE RNK = 1;

--TOP 5 CUSTEMORS BASED ON SALES
SELECT TOP 5
	CUSTOMER_ID,
	SUM(TOTAL_SALE) AS TOTAL_SALE
	FROM RETAIL_SALES_ANALYSIS
	GROUP BY CUSTOMER_ID
	ORDER BY TOTAL_SALE DESC ;
	

--UNIQUE CUSTOMERS WHO PURCHASED ITEM FORM EACH CATEGORY

SELECT 
	CATEGORY,
	COUNT(DISTINCT CUSTOMER_ID) AS UNIQUE_CUSTOMERS
	FROM RETAIL_SALES_ANALYSIS
	GROUP BY CATEGORY;

--ORDERS IN DIFFERENT TIMES

WITH HOURLY_SALE AS
(
SELECT *,
	CASE 
	WHEN DATEPART(HOUR, SALE_TIME) <12 THEN 'MORNING'
	WHEN DATEPART(HOUR, SALE_TIME) BETWEEN 12 AND 17 THEN 'AFTERNOON'
	ELSE 'EVENING'
	END AS SHIFT 
	FROM RETAIL_SALES_ANALYSIS
)
SELECT 
	SHIFT,
	COUNT(*) AS TOTAL_ORDERS
	FROM HOURLY_SALE
	GROUP BY SHIFT;
